# install packages  
`$ npm install`
# notes 
please add your database info to config file /config/config.json
# to create db
`$ npx sequelize-cli db:create`
# to migrate 
`$ npx sequelize-cli db:migrate`
# run seeds 
`$ npx sequelize-cli db:seed:all`
# run app 
`$ npm start`
